#include<iostream>
#include<vector>
using namespace std;
int n;
vector<int> ans,num01,num02;//num01记录dfs里面进一步阶乘的数字
vector<int> mul(vector<int> num1,vector<int> num2)
{
    vector<int> c(num1.size()+num2.size());
    for(int i = 0;i<num1.size();i++)
    {
        for(int j = 0;j<num2.size();j++)
        {
            c[i+j] += num1[i]*num2[j];
            if(c[i+j]>=10)
            {
                c[i+j+1] += c[i+j]/10;
                c[i+j]%=10;
            }
        }
    }
    while (!c.empty()&&c.back() == 0) c.pop_back();
    return c;
}
vector<int> add(vector<int> num1,vector<int> num2)
{
    vector<int> c;
    int yu = 0;
    for(int i = 0;i<max(num1.size(),num2.size());i++)
    {
        if(i<num1.size()) yu+=num1[i];
        if(i<num2.size()) yu+=num2[i];
        c.push_back(yu%10);
        yu/=10;
    }
    if(yu!=0) c.push_back(yu);
    return c;
}

vector<int> dfs(int i)//计算第几个阶乘
{
    vector<int> temp,gen;
    temp.push_back(1);
    while(i)
    {
        num01.clear();
        int k = i;
        while (k)
        {
            num01.push_back(k%10);
            k/=10;
        }
        temp = mul(num01,temp);
        i--;
    }
    return temp;
}

int main()
{
    ans.push_back(0);
    cin>>n;
    for(int i = n;i>=1;i--)//第几个阶乘
    {
        auto numre = dfs(i);
        ans = add(ans,numre);
    }
    for(int i = ans.size()-1;i>=0;i--) cout<<ans[i];
    return 0;
}
