[分组](https://www.luogu.com.cn/problem/P4447)  
对于每一个元素，如果可以加入已有组，就直接加入；否则，就新开一组。  
外层for遍历每一个数，内层for找到能够插入的组。  
问题1：时间复杂度是O(n^2)，会超时。  
方法：内层for修改为二分查找。  
问题2：如何表示1e5个组。
方法：用最后一个数表示改组（每组单调递增，公差为1），开两个数组表示当前组的最后一个数和大小，
int cnt记录组数。  
语法：upper_bound(a,a+n,num):a数组,0_base下,[0,n]内,第一个大于num的数 
再-a:第一个大于num的数的下标；  
再-1:第一个小于等于num的数的下标；

```
#include<iostream>
#include<algorithm>
using namespace std;
int q[100009];//k[i]分完后的每组的max
int siz[100009];//记录数组长度
int a[100009];//原有的成员
int n,cnt;
int main(){
     cin>>n;
     for(int i=1;i<=n;i++)cin>>a[i];
     sort(a+1,a+n+1);
     q[0]=2e9;
     for(int i=1;i<=n;i++)
     {
        int k=upper_bound(q+1,q+cnt+1,a[i]-1)-q-1;
        if(q[k]==a[i]-1)q[k]=a[i],siz[k]++;
        else q[++cnt]=a[i],siz[cnt]=1;
     }
     int ans=1e9;
     for(int i=1;i<=cnt;i++)ans=min(ans,siz[i]);
     cout<<ans;
     return 0;
}
```
