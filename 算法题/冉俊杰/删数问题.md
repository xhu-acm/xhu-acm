//确保每次加入某元素，使当前数值最小。结果就最小；
//如何确保？用栈！
//对于一个元素b，在可删处除范围k内，推出小于该元素b的所有元素a，并将该元素b压入；
//保证了高位最小;从而保证了当前数字最小；
#include<iostream>
#include<stack>
#include<algorithm>
using namespace std;
int main()
{
    string n;
    int k;
    stack<char>st;
    cin>>n>>k;

    for(char c:n)
    {
        while(k>0&&!st.empty()&&c<st.top())
        {
            st.pop();
            k--;
        }
        st.push(c);
    }
    while(k>0&&!st.empty())st.pop(),k--;

    string re="";
    while(!st.empty())
    {
        re+=st.top();
        st.pop();
    }
    reverse(re.begin(),re.end());
    int start=0;
    while(re[start]=='0'&&start<re.size()-1)start++;

    cout<<re.substr(start);
    return 0;
}
